// 数据存储，c++的数据存储方式是通过存储持续性、作用域、链接性来描述的

// 1. 主要按存储持续性分类(c++11中是四种)
```cpp
// (1). 自动存储持续性(自动变量，存在栈，后进先出),随着代码段的调用而开始和结束
void printt(int i){
    int b=0; // 自动变量一定要初始化，不然会是不确定性，导致生命周期混乱，影响内存。
cout<<i<<endl; // 局部变量
}

// (2). 静态存储持续性(静态变量，存在静态区)，全局存在，直至程序结束，如果没有
int a; // 链接性为外部的静态变量（全局变量），其他文件也能调用
static int i; // 链接性为内部的静态变量，只能在本文件调用
void printt(){
    static int b; // 无链接性的静态变量，只能在本代码段调用
}

// (3). 线程存储持续性(线程局部变量，存在？)，与线程的生命线一致
thread_local int a=1;
void setThreadVar(){a=2;}

int main(){
std::thread t1(setThreadVar); // 新起一个线程
t1.join();
cout<<a<<endl; // 1,主线程的线程变量没变，不同线程之间的线程变量互不干扰
}

// (4).动态存储持续性(动态变量，存在堆，后进后出)，由程序员自己控制
int *i=new int{42};
delete i; // 有new就一定要有delete，不然容易内存泄漏。
```
